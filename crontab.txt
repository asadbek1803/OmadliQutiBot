# Django crontab configuration
# Add to your crontab with: crontab -e

# Option 1: Combined server + maintenance (recommended)
# Runs server with built-in daily maintenance at 00:00
@reboot cd "C:\Users\Lenovo\Desktop\OmadliQuti" && python manage.py runserver_with_maintenance --port 8000

# Option 2: Separate server and maintenance
# Start server on reboot
@reboot cd "C:\Users\Lenovo\Desktop\OmadliQuti" && python manage.py runserver 0.0.0.0:8000

# Daily maintenance at 00:00
0 0 * * * cd "C:\Users\Lenovo\Desktop\OmadliQuti" && python manage.py daily_maintenance_with_scheduler

# Option 3: Manual start with maintenance

# python manage.py runserver_with_maintenance --port 8000 --maintenance-time 00:00

# Option 4: Backup only every 6 hours
0 */6 * * * cd "C:\Users\Lenovo\Desktop\OmadliQuti" && python manage.py daily_maintenance --backup-only
